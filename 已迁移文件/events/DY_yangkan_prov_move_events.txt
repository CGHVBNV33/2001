#id = 1 : 主系统-幕后执行的
#id = 2 : 羊侃｢【武将】 毛遂自荐 
namespace = yangkan_prov_move_events
province_event = {
	id = yangkan_prov_move_events.1
	title = "yangkan_prov_move_events.NAME1"
	desc = "yangkan_prov_move_events.DESC1"
	picture = RELIGIOUS_WARS_eventPicture
	hidden = yes
	trigger = {
		#侦测省份内是否有该pov_xxxx
		#只要有新增的武将，记得加入该pov_xxxx 于 OR 其中
		has_province_modifier = pov_yangkan
	}
	mean_time_to_happen = {
		months = 2
	}
	option = {
		name = "yangkan_prov_move_events.OPTA1"
		###########################羊侃｢【武将】pov_yangkan#################################		
		if = {
			limit = {
				has_province_modifier = pov_yangkan
			}
			if = {
				#决定在武将的寿命
				limit = {
					NOT = {
						is_year = 549
					}
				}
				#假如该国家的君王特质符合
				#就毛遂自荐	
				if = {
					limit = {
						owner = {
							NOT = {
								tag = JAP
							}
							OR = {
								has_ruler_modifier = ra_ch1
								has_ruler_modifier = ra_ch2
								has_ruler_modifier = ra_ch6
								has_ruler_modifier = ra_ch8
								has_ruler_modifier = rb_ch4
								has_ruler_modifier = rb_ch6
							}
						}
					}
					remove_province_modifier = pov_yangkan
					owner = {
						country_event = {
							id = yangkan_prov_move_events.2
						}
					}
				}
				#假如该国家的君王特质不符合就到邻近的省分						
				if = {
					limit = {
						owner = {
							NOT = {
								tag = JAP
							}
							NOT = {
								OR = {
									has_ruler_modifier = ra_ch1
									has_ruler_modifier = ra_ch2
									has_ruler_modifier = ra_ch6
									has_ruler_modifier = ra_ch8
									has_ruler_modifier = rb_ch4
									has_ruler_modifier = rb_ch6
								}
							}
						}
					}
					remove_province_modifier = pov_yangkan
					random_neighbor_province = {
						add_province_modifier = {
							name = "pov_yangkan"
							duration = -1
						}
					}
				}
				#不是期待的国家 直接移到别的省分
				if = {
					limit = {
						owner = {
							tag = JAP
						}
					}
					remove_province_modifier = pov_yangkan
					random_neighbor_province = {
						add_province_modifier = {
							name = "pov_yangkan"
							duration = -1
						}
					}
				}
			}
			if = {
				#如果年份超过此限，在野武将离世
				limit = {
					is_year = 549
				}
				remove_province_modifier = pov_yangkan
			}
		}
	}
}
