#id = 1 : 主系統-幕後執行的
#id = 2 : 伊墨居次｢【军师】 毛遂自薦 

namespace = yimojuci_prov_move_events


province_event = {
	id = yimojuci_prov_move_events.1
	title = "yimojuci_prov_move_events.NAME1"
	desc = "yimojuci_prov_move_events.DESC1"
	picture = RELIGIOUS_WARS_eventPicture
	
	hidden = yes

	trigger = {
	#偵測省份內是否有該pov_xxxx
	#只要有新增的武將，記得加入該pov_xxxx 於 OR 其中
		has_province_modifier = pov_yimojuci
	}
	
	
	mean_time_to_happen = {
		months = 12
	}
	
	option = {
		name = "yimojuci_prov_move_events.OPTA1"
###########################伊墨居次｢【军师】pov_yimojuci#################################		
		if = {
			limit = { has_province_modifier = pov_yimojuci }
			if = {
			#決定在野武將的壽命
				limit = { 
					NOT = { is_year = 30  }
				}
			#假如該國家的君王特質符合	
			#就毛遂自薦	
				if = { 
					limit = {
						owner = { 
							NOT = { tag = JAP } 
								OR = {
									has_ruler_modifier = ra_ch1
									has_ruler_modifier = ra_ch3
									has_ruler_modifier = ra_ch6
									has_ruler_modifier = ra_ch8
									has_ruler_modifier = rb_ch7
									has_ruler_modifier = rb_ch8
														
								}
							}
						}
			
					remove_province_modifier = pov_yimojuci
					owner = {
						country_event = {
							id = yimojuci_prov_move_events.2
						}
					}
				}
					#假如該國家的君王特質不符合就到鄰近的省分						
				if = { 
					limit = { 
						owner = {
							NOT = { tag = JAP } 
							NOT = { 
								OR = {
								    has_ruler_modifier = ra_ch1
								    has_ruler_modifier = ra_ch3
									has_ruler_modifier = ra_ch6
									has_ruler_modifier = ra_ch8
									has_ruler_modifier = rb_ch7
									has_ruler_modifier = rb_ch8
            
								}
							}
						}		
					}
					remove_province_modifier = pov_yimojuci
					random_neighbor_province = {
						add_province_modifier = {
							name = "pov_yimojuci"
							duration = -1
						}					
					}
				}
				if = { 
					limit = { 
						owner = { tag = JAP } 
					}
					remove_province_modifier = pov_yimojuci
					random_neighbor_province = {
						add_province_modifier = {
							name = "pov_yimojuci"
							duration = -1
						}					
					}								
				}
			}	
			if = {
			#如果年份超過此限，在野武將離世
				limit = { 
					is_year = 30 
				}	
				remove_province_modifier = pov_yimojuci		
			}
		}
	}
}

				fire = 4
				shock = 5

				siege = 2
				trait = rab_leader_yimojuci
			}
		}
	}
