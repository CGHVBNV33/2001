#id = 1 : 主系统-幕后执行的
#id = 2 : 博尔朮｢【武将】毛遂自荐 

namespace = boershu_prov_move_events


province_event = {
	id = boershu_prov_move_events.1
	title = "boershu_prov_move_events.NAME1"
	desc = "boershu_prov_move_events.DESC1"
	picture = RELIGIOUS_WARS_eventPicture
	
	hidden = yes

	trigger = {
	#侦测省份内是否有该pov_xxxx
		has_province_modifier = pov_boershu
	}
	
	
	mean_time_to_happen = {
		months = 2
	}
	
	option = {
		name = "boershu_prov_move_events.OPTA1"
###########################博尔朮｢【武将】pov_boershu#################################		
		if = {
			limit = { has_province_modifier = pov_boershu }
			if = {
			#决定在野武将的寿命
				limit = { 
					NOT = { is_year = 1248  }
				}
			#假如该国家的君王特质符合
			#就毛遂自荐	
				if = { 
					limit = {
						owner = { 
							NOT = {
								OR = {
									tag = JAP 
									tag = ENG
								}
							} 
								OR = {
					
						has_ruler_modifier = ra_ch1
						has_ruler_modifier = ra_ch2
						has_ruler_modifier = ra_ch5
						has_ruler_modifier = ra_ch6
						has_ruler_modifier = rb_ch2
						has_ruler_modifier = rb_ch3
						has_ruler_modifier = rb_ch4	
						has_ruler_modifier = rb_ch5
						has_ruler_modifier = rb_ch6
						has_ruler_modifier = rb_ch8
						
								}
							}
						}
			
					remove_province_modifier = pov_boershu
					owner = {
						country_event = {
							id = boershu_prov_move_events.2
						}
					}
				}
					#假如该国家的君王特质不符合就到邻近的省分						
				if = { 
					limit = { 
						owner = {
							NOT = {
								OR = {
									tag = JAP 
									tag = ENG
								}
							} 
							NOT = { 
								OR = {
						has_ruler_modifier = ra_ch1
						has_ruler_modifier = ra_ch2
						has_ruler_modifier = ra_ch5
						has_ruler_modifier = ra_ch6
						has_ruler_modifier = rb_ch2
						has_ruler_modifier = rb_ch3
						has_ruler_modifier = rb_ch4	
						has_ruler_modifier = rb_ch5
						has_ruler_modifier = rb_ch6
						has_ruler_modifier = rb_ch8
								}
							}
						}		
					}
					remove_province_modifier = pov_boershu
					random_neighbor_province = {
						add_province_modifier = {
							name = "pov_boershu"
							duration = -1
						}					
					}
				}
				if = { 
					limit = { 
						owner = {
							OR = {
								tag = JAP 
								tag = ENG
							}
						} 
					}
					remove_province_modifier = pov_boershu
					random_neighbor_province = {
						add_province_modifier = {
							name = "pov_boershu"
							duration = -1
						}					
					}								
				}
			}	
			if = {
			#如果年份超过此限，在野武将离世
				limit = { 
					is_year = 1248 
				}	
				remove_province_modifier = pov_boershu		
			}
		}
	}
}

			}
		}
	}
