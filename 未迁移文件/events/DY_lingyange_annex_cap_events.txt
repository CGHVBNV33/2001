country_event = {
	id = zhanggongjin_annex_cap_events.2
	title = "zhanggongjin_annex_cap_events.NAME2"
	desc = "zhanggongjin_annex_cap_events.DESC2"
	picture = CAPTURE_GENERAL_eventPicture
	
	is_triggered_only = yes
	# major = yes 就是昭告天下
	major = yes
	
	option = {
		name = "zhanggongjin_annex_cap_events.OPTA2"
		#ai_chance 代表的是该选项AI会用多少的概率选择，这个分母加总于所有options
		ai_chance = {
			factor = 70
			modifier = {
				factor = 0 #不符合以下的条件，直接从70%掉到0%
				NOT = {
					OR = {
					
						has_ruler_modifier = ra_ch1		
						has_ruler_modifier = ra_ch2
						has_ruler_modifier = ra_ch4
						has_ruler_modifier = ra_ch5
						has_ruler_modifier = ra_ch7
						has_ruler_modifier = ra_ch8
						has_ruler_modifier = rb_ch5
						has_ruler_modifier = rb_ch6
						has_ruler_modifier = rb_ch7
													
					}
				}
			}
			modifier = {
				factor = 0 #符合以下的条件，直接从70%掉到0%
				OR = {
					tag = JAP
					tag = USA
				}	
			}	
		}
		
		if = {
			limit = {
				NOT = {
			 		OR = {
						tag = JAP
						tag = USA
					}	
				}
				OR = {
					has_ruler_modifier = ra_ch1		
					has_ruler_modifier = ra_ch2
					has_ruler_modifier = ra_ch4
					has_ruler_modifier = ra_ch5
					has_ruler_modifier = ra_ch7
					has_ruler_modifier = ra_ch8
					has_ruler_modifier = rb_ch5
					has_ruler_modifier = rb_ch6
					has_ruler_modifier = rb_ch7		
			    }
			}
			
			if = { 
				limit = { ai = no }
					define_advisor = {
						type = zhanggongjin
						name = "张公谨｢【武将】"
						discount = yes	
						skill = 5
					}
			}
			if = { 
				limit = { ai = yes }
					define_general = {
					name = "张公谨｢【武将】"
					fire = 4
					shock = 4
					manuever = 3
					siege = 3
					trait = rab_leader_zhanggongjin
				}
			}
		}
		else = {
		#玩家登用，但是君王or other条件不符合
		#武将拒绝的画
			country_event = {
				id = zhanggongjin_annex_cap_events.3
			}		
		}		
	}

	option = {
		#AI或玩家砍了他
		#	
		name = "zhanggongjin_annex_cap_events.OPTB2"
		ai_chance = { factor = 10 }	
		add_prestige = -2
		add_treasury = 100
		
	}	

	option = {
		#AI或玩家放他走了
		#武将将会开始流浪
		name = "zhanggongjin_annex_cap_events.OPTC2"
		ai_chance = { factor = 20 }
		add_prestige = 2
		hidden_effect = { 
			random_owned_province = {
				add_province_modifier = {
					name = "pov_zhanggongjin"
					duration = -1
				}
			}
		}
	}	
}
