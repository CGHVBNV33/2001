#id = 1 : 主系統-幕後執行的
#id = 2 : 王浚｢【武将】毛遂自薦 

namespace = wangjun_prov_move_events


province_event = {
	id = wangjun_prov_move_events.1
	title = "wangjun_prov_move_events.NAME1"
	desc = "wangjun_prov_move_events.DESC1"
	picture = RELIGIOUS_WARS_eventPicture
	
	hidden = yes

	trigger = {
	#偵測省份內是否有該pov_xxxx
	#只要有新增的武將，記得加入該pov_xxxx 於 OR 其中
		has_province_modifier = pov_wangjun
	}
	
	
	mean_time_to_happen = {
		months = 24
	}
	
	option = {
		name = "wangjun_prov_move_events.OPTA1"
###########################王浚｢【武将】pov_wangjun#################################		
		if = {
			limit = { has_province_modifier = pov_wangjun }
			if = {
			#決定在野武將的壽命
				limit = { 
					NOT = { is_year = 334  }
				}
			#假如該國家的君王特質符合	
			#就毛遂自薦	
				if = { 
					limit = {
						owner = { 
							NOT = { tag = JAP } 
								OR = {
									has_ruler_modifier = ra_ch1
									has_ruler_modifier = ra_ch3
									has_ruler_modifier = ra_ch6
									has_ruler_modifier = ra_ch9
									has_ruler_modifier = rb_ch5
									has_ruler_modifier = rb_ch6
														
								}
							}
						}
			
					remove_province_modifier = pov_wangjun
					owner = {
						country_event = {
							id = wangjun_prov_move_events.2
						}
					}
				}
					#假如該國家的君王特質不符合就到鄰近的省分						
				if = { 
					limit = { 
						owner = {
							NOT = { tag = JAP } 
							NOT = { 
								OR = {
								    has_ruler_modifier = ra_ch1
								    has_ruler_modifier = ra_ch3
									has_ruler_modifier = ra_ch6
									has_ruler_modifier = ra_ch9
									has_ruler_modifier = rb_ch5
									has_ruler_modifier = rb_ch6
            
								}
							}
						}		
					}
					remove_province_modifier = pov_wangjun
					random_neighbor_province = {
						add_province_modifier = {
							name = "pov_wangjun"
							duration = -1
						}					
					}
				}
				if = { 
					limit = { 
						owner = { tag = JAP } 
					}
					remove_province_modifier = pov_wangjun
					random_neighbor_province = {
						add_province_modifier = {
							name = "pov_wangjun"
							duration = -1
						}					
					}								
				}
			}	
			if = {
			#如果年份超過此限，在野武將離世
				limit = { 
					is_year = 334 
				}	
				remove_province_modifier = pov_wangjun		
			}
		}
	}
}
