namespace = linrenzhao_prov_move_events


province_event = {
	id = linrenzhao_prov_move_events.1
	title = "linrenzhao_prov_move_events.NAME1"
	desc = "linrenzhao_prov_move_events.DESC1"
	picture = RELIGIOUS_WARS_eventPicture
	
	hidden = yes

	trigger = {
	#偵測省份內是否有該pov_xxxx
		has_province_modifier = pov_linrenzhao
	}
	
	
	mean_time_to_happen = {
		months = 5
	}
	
	option = {
		name = "linrenzhao_prov_move_events.OPTA1"
##################################################		
		if = {
			limit = { has_province_modifier = pov_linrenzhao }
			if = {
			#
				limit = { 
					NOT = { is_year = 972  }
				}
			#
			#
				if = { 
					limit = {
						owner = { 
							NOT = {
								OR = {
									tag = SNG
								}
							} 
							OR = {
								has_ruler_modifier = ra_ch1		
								has_ruler_modifier = ra_ch2
								has_ruler_modifier = ra_ch4
								has_ruler_modifier = ra_ch5
								has_ruler_modifier = ra_ch7
								has_ruler_modifier = ra_ch8
								has_ruler_modifier = rb_ch5
								has_ruler_modifier = rb_ch6
								has_ruler_modifier = rb_ch7
													
							}
						}
					}
			
					remove_province_modifier = pov_linrenzhao
					owner = {
						country_event = {
							id = linrenzhao_prov_move_events.2
						}
					}
				}
					#假如						
				if = { 
					limit = { 
						owner = {
							NOT = {
								OR = {
									tag = SNG
								}
							} 
							NOT = {
								OR = {
									has_ruler_modifier = ra_ch1		
									has_ruler_modifier = ra_ch2
									has_ruler_modifier = ra_ch4
									has_ruler_modifier = ra_ch5
									has_ruler_modifier = ra_ch7
									has_ruler_modifier = ra_ch8
									has_ruler_modifier = rb_ch5
									has_ruler_modifier = rb_ch6
									has_ruler_modifier = rb_ch7
								}		
							}
						}		
					}
					remove_province_modifier = pov_linrenzhao
					random_neighbor_province = {
						add_province_modifier = {
							name = "pov_linrenzhao"
							duration = -1
						}					
					}
				}
				if = { 
					limit = { 
						owner = {
							OR = {
								tag = SNG
							}
						} 
					}
					remove_province_modifier = pov_linrenzhao
					random_neighbor_province = {
						add_province_modifier = {
							name = "pov_linrenzhao"
							duration = -1
						}					
					}								
				}
			}	
			if = {
			#如果年份higher than below years, then it will clear the province modifier
				limit = { 
					is_year = 972
				}	
				remove_province_modifier = pov_linrenzhao		
			}
		}
	}
}
