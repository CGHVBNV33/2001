province_event = {
	id = charlie_chaplin_prov_move_events.1
	title = "charlie_chaplin_prov_move_events.NAME1"
	desc = "charlie_chaplin_prov_move_events.DESC1"
	picture = RELIGIOUS_WARS_eventPicture
	
	hidden = yes

	trigger = {
		has_province_modifier = pov_charlie_chaplin
	}
	
	
	mean_time_to_happen = {
		months = 24
	}
	
	option = {
		name = "charlie_chaplin_prov_move_events.OPTA1"
	
		if = {
			limit = { has_province_modifier = pov_charlie_chaplin }
			if = {
			limit = { 
				NOT = { is_year = 1977  }
			}	
				if = { 
					limit = {
						owner = { 
							NOT = { tag = JAP } 
							OR = {
								ruler_has_personality = well_connected_personality
								ruler_has_personality = well_advised_personality
								ruler_has_personality = silver_tongue_personality
								ruler_has_personality = fierce_negotiator_personality
								ruler_has_personality = humane_personality
								ruler_has_personality = benevolent_personality				
							}
						}
					}
					remove_province_modifier = pov_charlie_chaplin
					owner = {
						country_event = {
							id = charlie_chaplin_prov_move_events.2
						}
					}
				}					
				if = { 
					limit = { 
						owner = {
							NOT = { tag = JAP } 
							NOT = { 
								OR = {
								  ruler_has_personality = well_connected_personality
							 	  ruler_has_personality = well_advised_personality
								  ruler_has_personality = silver_tongue_personality
								  ruler_has_personality = fierce_negotiator_personality
								  ruler_has_personality = humane_personality
								  ruler_has_personality = benevolent_personality
								}
							}
						}		
					}
					remove_province_modifier = pov_charlie_chaplin
					random_neighbor_province = {
						add_province_modifier = {
							name = "pov_charlie_chaplin"
							duration = -1
						}					
					}
				}
				if = { 
					limit = { 
						owner = { tag = JAP } 
					}
					remove_province_modifier = pov_charlie_chaplin
					random_neighbor_province = {
						add_province_modifier = {
							name = "pov_charlie_chaplin"
							duration = -1
						}					
					}								
				}
			}		
			if = {
			limit = { 
				is_year = 1977
			}	
				remove_province_modifier = pov_charlie_chaplin	
			}
		}
	}	
}
