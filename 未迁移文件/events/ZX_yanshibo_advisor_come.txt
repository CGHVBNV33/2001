country_event = {
	id = yanshibo_advisor_come.1
	title = "yanshibo_advisor_come.1.t"

	picture = NEW_ADVISOR_eventPicture
	#隱藏事件

		is_year = 441

			AND = {
				exists = LSO
				tag = LSO
			}
			AND = {

		}
		}
	}

	mean_time_to_happen = {
		days = 5
	}

		name = yanshibo_advisor_come.OPTA1
		if = { 
		limit = { ai = no }
			define_advisor = {
				type = yanshibo
				name = "颜师伯｢【武将】"
				discount = yes	#顾问打折
				skill = 3		#顾问等级
				female = no #如果你創建的是女的 請改成yes
			}
		}
		#須加上｢ 
		if = { 
		limit = { ai = yes }
			define_general = {
				name = "颜师伯｢【武将】"
				fire = 5
				shock = 6
				manuever = 4
				siege = 3
				trait = rab_leader_yanshibo #将领的特质
			}
		}
	}
}
