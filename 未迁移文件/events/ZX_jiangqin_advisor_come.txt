#武将某种條件下出現
#id = 1 : jiangqin : 蒋钦｢【武将】

country_event = {
	id = jiangqin_advisor_come.1
	title = "jiangqin_advisor_come.1.t"
	desc = "jiangqin_advisor_come.1.d"
	picture = NEW_ADVISOR_eventPicture
	#隱藏事件
	hidden = yes
	#整個遊戲只能觸發一次
	fire_only_once = yes


	trigger = {
	#武将出现在顾问池内的时间点和地点 或国家tag
		is_year = 191
		NOT = { is_year = 219 }
		OR = {
			AND = {
				NOT = { exists = SNC }
				exists = WUU
				tag = WUU
			}
			AND = {
				NOT = { exists = WUU }
				exists = SNC
				tag = SNC
			}
			
			AND = {
				NOT = { exists = SNC }
				NOT = { exists = WUU }
				owns = 5042
			}
		}
	}
	#觸發時間沒差1-5days均可
	mean_time_to_happen = {
		days = 5
	}		
	option = {
		name = jiangqin_advisor_come.OPTA1
		if = { 
		limit = { ai = no }
			define_advisor = {
				type = jiangqin
				name = "蒋钦｢【武将】"
				discount = yes	#顾问打折
				skill = 3		#顾问等级
				female = no #如果你創建的是女的 請改成yes
			}
		}
		#須加上｢ 
		if = { 
		limit = { ai = yes }
			define_general = {
				name = "蒋钦｢【武将】"
				fire = 3
				shock = 4
				manuever = 3
				siege = 2
				trait = rab_leader_jiangqin #将领的特质
			}
		}
	}
}
