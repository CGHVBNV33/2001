namespace = andrea_doria_prov_move_events

province_event = {
	id = andrea_doria_prov_move_events.1
	title = "andrea_doria_prov_move_events.NAME1"
	desc = "andrea_doria_prov_move_events.DESC1"
	picture = RELIGIOUS_WARS_eventPicture
	
	hidden = yes

	trigger = {
		has_province_modifier = pov_andrea_doria
	}
	
	
	mean_time_to_happen = {
		months = 2
	}
	
	option = {
		name = "andrea_doria_prov_move_events.OPTA1"		
		if = {
			limit = { has_province_modifier = pov_andrea_doria }
			if = {
				limit = { 
					NOT = { is_year = 1560 }
				}
				if = { 
					limit = {
						owner = { 
							NOT = {
								OR = {
									tag = JAP 
									tag = JA1
								}
							}
							OR = {
								ruler_has_personality = well_connected_personality
								ruler_has_personality = well_advised_personality
								ruler_has_personality = silver_tongue_personality
								ruler_has_personality = fierce_negotiator_personality
								ruler_has_personality = humane_personality
								ruler_has_personality = benevolent_personality				
							}
						}
					}	
					remove_province_modifier = pov_andrea_doria
					owner = {
						country_event = {
							id = andrea_doria_prov_move_events.2
						}
					}
				}					
				if = { 
					limit = { 
						owner = {
							OR = {
								tag = JA1
								tag = JAP
							}	
							NOT = { 
								OR = {
								 	ruler_has_personality = well_connected_personality
							 	  	ruler_has_personality = well_advised_personality
								  	ruler_has_personality = silver_tongue_personality
								  	ruler_has_personality = fierce_negotiator_personality
								  	ruler_has_personality = humane_personality
								  	ruler_has_personality = benevolent_personality
								}
							}
						}		
					}
					remove_province_modifier = pov_andrea_doria
					random_neighbor_province = {
						add_province_modifier = {
							name = "pov_andrea_doria"
							duration = -1
						}					
					}
				}
			}	
			if = {
				limit = { 
					is_year = 1560
				}	
				remove_province_modifier = pov_andrea_doria	
			}
		}
	}
}
