namespace = tadeusz_kosciuszko_prov_move_events

province_event = {
	id = tadeusz_kosciuszko_prov_move_events.1
	title = "tadeusz_kosciuszko_prov_move_events.NAME1"
	desc = "tadeusz_kosciuszko_prov_move_events.DESC1"
	picture = RELIGIOUS_WARS_eventPicture
	
	hidden = yes

	trigger = {
		has_province_modifier = pov_tadeusz_kosciuszko
	}
	
	
	mean_time_to_happen = {
		months = 2
	}
	
	option = {
		name = "tadeusz_kosciuszko_prov_move_events.OPTA1"		
		if = {
			limit = { has_province_modifier = pov_tadeusz_kosciuszko }
			if = {
				limit = { 
					NOT = { is_year = 1746 }
				}
				if = { 
					limit = {
						owner = { 
							NOT = {
								OR = {
									tag = JAP 
									tag = JA1
								}
							}
							OR = {
								ruler_has_personality = well_connected_personality
								ruler_has_personality = well_advised_personality
								ruler_has_personality = silver_tongue_personality
								ruler_has_personality = fierce_negotiator_personality
								ruler_has_personality = humane_personality
								ruler_has_personality = benevolent_personality				
							}
						}
					}	
					remove_province_modifier = pov_tadeusz_kosciuszko
					owner = {
						country_event = {
							id = tadeusz_kosciuszko_prov_move_events.2
						}
					}
				}					
				if = { 
					limit = { 
						owner = {
							OR = {
								tag = JA1
								tag = JAP
							}	
							NOT = { 
								OR = {
								 	ruler_has_personality = well_connected_personality
							 	  	ruler_has_personality = well_advised_personality
								  	ruler_has_personality = silver_tongue_personality
								  	ruler_has_personality = fierce_negotiator_personality
								  	ruler_has_personality = humane_personality
								  	ruler_has_personality = benevolent_personality
								}
							}
						}		
					}
					remove_province_modifier = pov_tadeusz_kosciuszko
					random_neighbor_province = {
						add_province_modifier = {
							name = "pov_tadeusz_kosciuszko"
							duration = -1
						}					
					}
				}
			}	
			if = {
				limit = { 
					is_year = 1817
				}	
				remove_province_modifier = pov_tadeusz_kosciuszko	
			}
		}
	}
}
